This module implements a generative function combinator which allows statically unrolled control flow for generative functions which can act as kernels -- mapping over vectorial versions of their arguments.

 41@dataclass
 42class MapTrace(Trace):
 43    gen_fn: GenerativeFunction

 45    args: Tuple

 48    def get_args(self):
 49        return self.args

 55        return self.gen_fn

 57    def get_retval(self):
 58        return self.mapped_subtrace.get_retval()

 60    def get_score(self):
 61        return self.score

 80@dataclass
 81class MapCombinator(GenerativeFunction):
 82    kernel: GenerativeFunction
 83
 84    def flatten(self):
 85        return (), (self.kernel,)
 86
 87    @classmethod
 88    def unflatten(cls, data, xs):
 89        return MapCombinator(*data, *xs)
 90
 91    def __call__(self, key, *args, **kwargs):
 92        in_axes = kwargs["in_axes"]
 93        vmapped = jax.vmap(sample(self.kernel), in_axes=in_axes)
 94        return vmapped(key, args)
 95
 96    from functools import partial
 97
 98    def simulate(self, key, args, **kwargs):
 99        in_axes = kwargs["in_axes"]
100        key, tr = jax.vmap(simulate(self.kernel), in_axes=in_axes)(key, args)
101        map_tr = MapTrace(
102            self,
103            tr,
104            args,
105            jnp.sum(tr.get_score()),
106        )
107
108        return key, map_tr
109
110    def importance(self, key, chm, args, **kwargs):
111        in_axes = kwargs["in_axes"]
112        key, (w, tr) = jax.vmap(importance(self.kernel), in_axes=in_axes)(
113            key, chm, args
114        )
115        map_tr = MapTrace(
116            self,
117            tr,
118            args,
119            jnp.sum(tr.get_score()),
120        )
121
122        return key, (w, map_tr)
123
124    def diff(self, key, original, new, args, **kwargs):
125        in_axes = kwargs["in_axes"]
126        key, (w, tr) = jax.vmap(diff(self.kernel), in_axes=in_axes)(
127            key, original, new, args
128        )
129        map_tr = MapTrace(
130            self,
131            tr,
132            args,
133            jnp.sum(tr.get_score()),
134        )
135
136        return key, (w, map_tr)
137
138    def update(self, key, original, new, args, **kwargs):
139        in_axes = kwargs["in_axes"]
140        key, (w, tr, discard) = jax.vmap(update(self.kernel), in_axes=in_axes)(
141            key, original, new, args
142        )
143        map_tr = MapTrace(
144            self,
145            tr,
146            args,
147            jnp.sum(tr.get_score()),

149
150        return key, (w, map_tr, discard)
151
152    def arg_grad(self, key, tr, args, **kwargs):
153        in_axes = kwargs["in_axes"]
154        key, arg_grads = jax.vmap(arg_grad(self.kernel), in_axes=in_axes)(
155            key, tr, args
156        )
157        return key, arg_grads
158
159    def choice_grad(self, key, tr, chm, args, **kwargs):
160        in_axes = kwargs["in_axes"]
161        key, choice_grads = jax.vmap(choice_grad(self.kernel), in_axes=in_axes)(
162            key, tr, chm, args
163        )
164        return key, choice_grads

277class partial:
278    """New function with partial application of the given arguments
279    and keywords.
280    """
281
282    __slots__ = "func", "args", "keywords", "__dict__", "__weakref__"
283
284    def __new__(cls, func, /, *args, **keywords):
285        if not callable(func):
286            raise TypeError("the first argument must be callable")
287
288        if hasattr(func, "func"):
289            args = func.args + args
290            keywords = {**func.keywords, **keywords}
291            func = func.func
292
293        self = super(partial, cls).__new__(cls)
294
295        self.func = func
296        self.args = args
297        self.keywords = keywords
298        return self
299
300    def __call__(self, /, *args, **keywords):
301        keywords = {**self.keywords, **keywords}
302        return self.func(*self.args, *args, **keywords)
303
304    @recursive_repr()
305    def __repr__(self):
306        qualname = type(self).__qualname__
307        args = [repr(self.func)]
308        args.extend(repr(x) for x in self.args)
309        args.extend(f"{k}={v!r}" for (k, v) in self.keywords.items())
310        if type(self).__module__ == "functools":
311            return f"functools.{qualname}({', '.join(args)})"
312        return f"{qualname}({', '.join(args)})"
313
314    def __reduce__(self):
315        return type(self), (self.func,), (self.func, self.args,
316               self.keywords or None, self.__dict__ or None)
317
318    def __setstate__(self, state):
319        if not isinstance(state, tuple):
320            raise TypeError("argument to __setstate__ must be a tuple")
321        if len(state) != 4:
322            raise TypeError(f"expected 4 items in state, got {len(state)}")
323        func, args, kwds, namespace = state
324        if (not callable(func) or not isinstance(args, tuple) or
325           (kwds is not None and not isinstance(kwds, dict)) or
326           (namespace is not None and not isinstance(namespace, dict))):
327            raise TypeError("invalid partial state")
328
329        args = tuple(args) # just in case it's a subclass
330        if kwds is None:
331            kwds = {}
332        elif type(kwds) is not dict: # XXX does it need to be *exactly* dict?
333            kwds = dict(kwds)
334        if namespace is None:
335            namespace = {}
336
337        self.__dict__ = namespace
338        self.func = func
339        self.args = args
340        self.keywords = kwds
