import jax
import jax.numpy as jnp
import genjax
from genjax import tag, grab, stab

console = genjax.pretty()
def f(x):
    x = tag(x + 1, name="x")
    result = tag(x**2, name="result")
    return result


v, state = genjax.grab(f)(5.0)
