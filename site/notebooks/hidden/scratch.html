    import jax
import jax.numpy as jnp

import genjax

console = genjax.pretty()


def emits_cc_gen_fn(v):
    @genjax.gen
    def model():
        x = genjax.normal(jnp.sum(v), 1.0) @ "x"
        return x

    return model


@genjax.gen
def model():
    x = jnp.ones(5)
    gen_fn = emits_cc_gen_fn(x)
    v = gen_fn.inline()
    return (v, gen_fn)


key = jax.random.PRNGKey(314159)
key, _ = jax.jit(genjax.simulate(model))(key, ())
